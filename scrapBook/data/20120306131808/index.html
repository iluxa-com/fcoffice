解决多个Ajax并发导致返回数据不同步的问题

2011年3月4日 由 chunterg 留言 »

最近在做项目中做到同时向Twitter(T),Flickr(F),Youtube(Y)请求数据，并需要将三者数据同时通过showResult()显示出来的功能。

由于向T,F,Y请求的方式不一样，返回数据的速度自然是不同的，如图：

所以需要在Ajax的回调方法里加上判断三者数据是否都成功返回的方法，然后再继续下面的showResult()。

我们都知道javascript是单线程的，且没有sleep(暂停)功能，网上普通采用的方法是用setTimeout加setInterval进行回调，不过这种方法对定时的估计没法做到很精确.
于是我尝试了直接用status(状态)判断加递归做。

流程图：

不多说，看代码：

var T=new Object(),//Twitter对象
	F=new Object(),//Flickr对象
	Y=new Object(),//Youtube对象
	needRecursion="";//用于判断是否需要回调方法的变量
	//将三者的初始状态设为wait，表示需要发送请求
	T.status="wait",F.status="wait",Y.status="wait";
function requestTFY(){
	/*
	 * 请求Twitter数据
	 */
	if(T.status=='wait'){
 
		T.status="ready";//将状态设为ready避免回调时重新发送请求
 
		$.getJSON('http://search.twitter.com/search.json?rpp=30&amp;'+
		'q=web&amp;lang=all&amp;callback=?',
		function(data){
			if(data.results.length&gt;0){
				T.status="done";//将状态设为done表示数据返回成功
				T.text = data.results[0].id;
 
				//判断是否需要递归
				if(needRecursion){
					 requestTFY();
				}
			}
		})
	}
/*
 * 请求Flickr数据
 */
if(F.status=='wait'){
	F.status="ready";
	$.getJSON('http://api.flickr.com/services/rest/?_=1299205020298&amp;method'+
	'=flickr.photos.search&amp;api_key=f0393c432fbbbaa26b30369b43b8f547&amp;'+
	'text=web&amp;safe_search=1&amp;sort=date-posted-desc&amp;per_page=10&amp;'+
	'page=1&amp;format=json&amp;jsoncallback=?',function(data){
		if(data.photos.photo.length&gt;0){
			F.status="done";
			F.text = data.photos.photo[0].id;
			if(needRecursion){
				 requestTFY();
			}
		}
	})
}
/*
 * 请求Youtube数据
 */
if(Y.status=='wait'){
	Y.status="ready";
	$.getJSON('http://gdata.youtube.com/feeds/api/videos?_=1299205004713&amp;'+
	'q=web&amp;format=5&amp;orderby=published&amp;max-results=30&amp;v=2&amp;start-index=1'+
	'&amp;alt=jsonc&amp;callback=?',function(data){
		if(data.data.items.length&gt;0){
			Y.status="done";
			Y.text = data.data.items[0].id;
			if(needRecursion){
				 requestTFY();
			}
		}
	})
}
	/*
	 *判断是否需要显示结果，并设置回调变量
	 */
   if(T.status=="done"&amp;&amp;F.status=="done"&amp;&amp;Y.status=="done"){
			showResult();//显示结果
			needRecursion = false;
	}else{
		needRecursion = true;
	}
}
